rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // ===== HELPER FUNCTIONS =====
    function isSignedIn() {
      return request.auth != null;
    }

    function isSelf(uid) {
      return isSignedIn() && request.auth.uid == uid;
    }

    function isAdmin() {
      return isSignedIn() && (
        request.auth.token.admin == true ||
        request.auth.token.role == 'admin' ||
        request.auth.token.email == 'admin@gmail.com'
      );
    }

    function isChatParticipant(chatData) {
      return isSignedIn() && request.auth.uid in chatData.participants;
    }

    function parentChat(chatId) {
      return get(/databases/$(database)/documents/chats/$(chatId)).data;
    }

    function hasValidMessageBody() {
      return (
        request.resource.data.message is string &&
        request.resource.data.message.size() > 0
      ) || (
        request.resource.data.text is string &&
        request.resource.data.text.size() > 0
      );
    }

    function isValidPresenceWrite(uid) {
      return isSelf(uid) &&
        request.resource.data.keys().hasOnly([
          'isOnline',
          'lastSeen',
          'updatedAt',
          'role'
        ]) &&
        request.resource.data.isOnline is bool;
    }

    function isAgent() {
      return isSignedIn() && (
        exists(/databases/$(database)/documents/TravelAgents/$(request.auth.uid)) ||
        isAdmin()
      );
    }

    // ===== USER PROFILE RULES =====
    match /Users/{uid} {
      allow read: if true;
      allow create, update, delete: if isSelf(uid) || isAdmin();
    }

    match /Profiles/{uid} {
      allow read: if true;
      allow create, update, delete: if isSelf(uid) || isAdmin();
    }

    // ===== TRAVEL AGENT RULES =====
    match /TravelAgents/{agentId} {
      allow read: if true;
      allow create, update, delete: if isSelf(agentId) || isAdmin();
    }

    // ===== APPROVED USERS SUBCOLLECTION =====
    // This allows agents to manage their approved users group
    match /agents/{agentId}/approvedUsers/{userId} {
      // Read permissions:
      // - Agent can read their own approved users list
      // - User can check if they are approved by this agent
      // - Admin can read all
      allow read: if isSignedIn() && (
        isSelf(agentId) ||
        isSelf(userId) ||
        isAdmin()
      );
      
      // Create permissions:
      // - Only the agent can add users to their approved list
      // - Must include correct agentId and userId
      allow create: if isSignedIn() && (
        isSelf(agentId) || isAdmin()
      ) && request.resource.data.agentId == agentId &&
        request.resource.data.userId == userId;
      
      // Update/Delete permissions:
      // - Agent can update or remove users from their approved list
      allow update, delete: if isSignedIn() && (
        isSelf(agentId) || isAdmin()
      );
    }

    // ===== APPROVED USERS COLLECTION GROUP =====
    // This allows users to query across all agents to find who approved them
    match /{path=**}/approvedUsers/{userId} {
      // Users can query the collectionGroup to find all agents who approved them
      allow read: if isSignedIn() && (
        isSelf(userId) || isAdmin()
      );
    }

    // ===== SCHEDULES RULES =====
    match /Schedules/{scheduleId} {
      allow read: if isSignedIn() && (
        resource.data.agentId == request.auth.uid ||
        isAdmin()
      );
      allow create: if isSignedIn() &&
        request.resource.data.agentId == request.auth.uid;
      allow update, delete: if isSignedIn() && (
        resource.data.agentId == request.auth.uid ||
        isAdmin()
      );
    }

    // ===== REQUESTS RULES =====
    match /Requests/{requestId} {
      allow create: if isSignedIn() &&
        request.resource.data.pilgrimId == request.auth.uid;
      allow read: if isSignedIn() && (
        resource.data.pilgrimId == request.auth.uid ||
        resource.data.agentId == request.auth.uid ||
        isAdmin()
      );
      allow update, delete: if isSignedIn() && (
        resource.data.agentId == request.auth.uid ||
        resource.data.pilgrimId == request.auth.uid ||
        isAdmin()
      );
    }

    // ===== ADMIN RULES =====
    match /admin/{docId} {
      allow read: if isSignedIn();
      allow write: if isAdmin();
    }

    // ===== RULES COLLECTION (Agent CRUD, User Read-Only) =====
    match /rules/{ruleId} {
      // Everyone can read all rules
      allow read: if isSignedIn();
      
      // Travel agents and admins can create rules
      allow create: if isAgent() &&
        request.resource.data.createdBy == request.auth.uid &&
        request.resource.data.ruleText is string &&
        request.resource.data.ruleText.size() >= 10 &&
        request.resource.data.category is string &&
        request.resource.data.createdAt is timestamp;
      
      // Only the agent who created the rule (or admin) can update it
      allow update: if isSignedIn() &&
        (resource.data.createdBy == request.auth.uid || isAdmin()) &&
        request.resource.data.createdBy == resource.data.createdBy &&
        request.resource.data.ruleText is string &&
        request.resource.data.ruleText.size() >= 10 &&
        request.resource.data.category is string;
      
      // Only the agent who created the rule (or admin) can delete it
      allow delete: if isSignedIn() &&
        (resource.data.createdBy == request.auth.uid || isAdmin());
    }

    // ===== USER PERSONAL DATA RULES =====
    match /manage-documents/{uid}/entries/{entryId} {
      allow read, create, update, delete: if isSelf(uid);
    }

    match /user_journals/{uid}/entries/{entryId} {
      allow read, create, update, delete: if isSelf(uid);
    }

    match /user_expenses/{uid}/entries/{entryId} {
      allow read, create, update, delete: if isSelf(uid);
    }

    match /user_checklists/{uid}/items/{itemId} {
      allow read, create, update, delete: if isSelf(uid);
    }

    // ===== PRESENCE RULES =====
    match /presence/{uid} {
      allow read: if isSignedIn();
      allow create, update: if isValidPresenceWrite(uid);
      allow delete: if isSelf(uid) || isAdmin();
    }

    // ===== CHAT RULES (MAIN - NO DOCUMENT EXISTENCE CHECK) =====
    match /chats/{chatId} {
      // Read: only participants can read
      allow read: if isChatParticipant(resource.data);
      
      // Create: signed-in user, must be one of 2 participants
      allow create: if isSignedIn() && 
        request.resource.data.participants is list &&
        request.resource.data.participants.size() == 2 &&
        request.auth.uid in request.resource.data.participants;
      
      // Update: only participants can update
      allow update, delete: if isChatParticipant(resource.data);

      // ===== MESSAGES SUB-COLLECTION =====
      match /messages/{messageId} {
        // Read: only chat participants can read messages
        allow read: if isChatParticipant(parentChat(chatId));
        
        // Create: participant + must be sender + must have message body
        allow create: if isChatParticipant(parentChat(chatId)) &&
          request.resource.data.senderId == request.auth.uid &&
          hasValidMessageBody();
        
        // Update: preserve message content, allow status updates only
        allow update: if isChatParticipant(parentChat(chatId)) &&
          request.resource.data.senderId == resource.data.senderId &&
          request.resource.data.message == resource.data.message &&
          request.resource.data.text == resource.data.text &&
          request.resource.data.timestamp == resource.data.timestamp;
        
        // Delete: chat participants can delete
        allow delete: if isChatParticipant(parentChat(chatId));
      }
    }
  }
}
